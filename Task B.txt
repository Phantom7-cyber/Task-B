#include <stdio.h>

int main() {
    int N, i = 1, sum = 0;
    printf("Enter a number N: ");
    scanf("%d", &N);

    while (i <= N) {
        if (i % 2 == 0) {
            sum += i;
        }
        i++;
    }

    printf("Sum of even numbers from 1 to %d = %d\n", N, sum);
    return 0;
}
